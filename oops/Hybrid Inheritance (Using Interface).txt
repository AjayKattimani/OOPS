class A {
	void showA() {
		System.out.println("Class A");
	}
}

class B extends A {
	void showB() {
		System.out.println("Class B");
	}
}

interface C {
	void showC();
}

class D extends B implements C {
	public void showC() {
		System.out.println("Interface C");
	}
}

public class Main {
	public static void main(String[] args) {
		D obj = new D();
		obj.showA();
		obj.showB();
		obj.showC();
	}
}
